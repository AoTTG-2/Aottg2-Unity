class Main
{
# Steve (EPK) Festive Town Defence 

    Description = "Steve's Town Defence: Protect the town from the approaching Titans!";
    StartTitans = 3;
    AddTitansPerWave = 1;
    _MaxWaves = 20;
    RespawnOnWave = true;
    GradualSpawn = true;
    GradualSpawnTooltip = "Spawn new titans gradually over time. Helpful for reducing lag.";
    _currentWave = 0;

    TitanRevealer = true;
    TitanRevealerTooltip = "If enabled, Titans will be revealed with an Outline when the titans in the round reach the value displayed below.";
    TitanRevealerCount = 3;
    TitanRevealerCountTooltip = "What number of Titans left alive until Titans are revealed with an Outline.";

    BonusRound = false;
    BonusRoundTooltip = "If enabled, there will an extra wave after Wave 20...";

    SpawnMultiplyer = 1;
    SpawnMultiplyerTooltip = "What value to multiply the number of Special Titans that will spawn. I.e 1x, 2x, 3x.";

    _hasSpawned = false;
    _T_override = false;
    _titans = 0;
    _humans = 0;
    _playerShifters = 0;

    _hornPivot = null;
    _h_r = null;
    _h_l = null;
    _punkoidCandyCane = null;
    _sleigh = null;
    _sleighPos = null;
    _sleighdecor = null;
    _krampusHat = null;
    _tree = null;

    # Store titan view id mapped to designated costume.
    _netbuffer = Dict();
    
    function OnGameStart()
    {
        self._tree = Map.FindMapObjectsByComponent("TheChristmasTree").Get(0).GetComponent("TheChristmasTree");
        self._punkoidCandyCane = Map.FindMapObjectByID(1102);
        self._sleigh = Map.FindMapObjectByID(1230);
        self._sleighdecor = Map.FindMapObjectByID(1104);
        self._sleighPos = Map.FindMapObjectByID(1229);
        self._hornPivot = Map.FindMapObjectByID(1231);
        self._h_r = Map.FindMapObjectByID(1232);
        self._h_l = Map.FindMapObjectByID(1233);
        self._krampusHat = Map.FindMapObjectByID(1247);

        if (Network.IsMasterClient)
        {
            self.NextWave();
        }

        if (self.BonusRound)
        {
            self._MaxWaves = 21;
        }
    }

    function OnSecond()
    {
        health = self._tree.GetHealth();
        maxHealth = self._tree.GetMaxHealth();
        label = "Health: " + Convert.ToString(health) + " / " + Convert.ToString(maxHealth);
        label = UI.WrapStyleTag(label, "color", "#FF0000");
        label = "Titans Left: " + Convert.ToString(self._titans) + "  " + "Wave: " + Convert.ToString(self._currentWave) + String.Newline + label;

        UI.SetLabelAll("TopCenter", label);
    }

    function OnCharacterSpawn(character)
    {
        if (character.IsMine && character.Type == "Titan")
        {
            if (!self._T_override)
            {
                character.DetectRange = 45;
                character.MoveTo(Vector3(0, 0, 0), 75, false);
            } 
        }

        self.TryResolveBuffer(character);

        if (Network.IsMasterClient)
        {
            self._titans = Game.Titans.Count;
            self._humans = Game.Humans.Count;
            self._playerShifters = Game.PlayerShifters.Count;
            self.gameEnd();
        }
        return;
    }

    function OnCharacterDie(victim, killer, killerName)
    {
        if(self.TitanRevealer)
        {
            if(self.TitanRevealerCount >= self._titans)
            {
                for (Titans in Game.AITitans)
                {
                    Titans.Reveal(90.5);
                }
            }
        }

        self.TryCleanBuffer(victim);

        if (Network.IsMasterClient)
        {
        self._titans = Game.Titans.Count;
        self._humans = Game.Humans.Count;
        self._playerShifters = Game.PlayerShifters.Count;
        self.gameEnd();
        }
        return;
    }

    function OnPlayerLeave(player)
    {
        if (Network.IsMasterClient)
        {
            self._titans = Game.Titans.Count;
            self._humans = Game.Humans.Count;
            self._playerShifters = Game.PlayerShifters.Count;
            self.gameEnd();
        }
    }

    function gameEnd()
    {
        if (Network.IsMasterClient && !Game.IsEnding)
        {
            if (self._humans > 0 || self._playerShifters > 0)
            {
                self._hasSpawned = true;
            }
            if (self._titans == 0 && self._currentWave > self._MaxWaves)
            {
                    UI.SetLabelAll("MiddleCenter", "Humanity wins!");
                    Game.End(10.0);
                    return; 
            }
            if (self._titans == 0)
            {
                self.NextWave();
            }
            if (self._humans == 0 && self._playerShifters == 0 && self._hasSpawned)
            {
                UI.SetLabelAll("MiddleCenter", "Humanity has failed to defend Christmas!");
                Game.End(10.0);
                return;
            }
        }
    }

    function NextWave()
    {
        self._currentWave = self._currentWave + 1;
        amount = self.AddTitansPerWave * (self._currentWave - 1) + self.StartTitans;
        if (self._currentWave > self._MaxWaves)
        {
            UI.SetLabelAll("MiddleCenter", "All waves cleared, humanity wins!");
            Game.End(10.0);
            return;
        }
        if (self.RespawnOnWave)
        {
            Game.SpawnPlayerAll(false);
        }
        if (self._currentWave == 5)
        {
            numby = 4 * self.SpawnMultiplyer;
            self.steadySpawnPunk(numby);

            amount = amount / 2;
            fairAmount = Math.Ceil(amount);

            self.gradualSpawnCheck(fairAmount);
            return;
        }
        if (self._currentWave == 10)
        {
            numby = 6 * self.SpawnMultiplyer;
            self.steadySpawnPunk(numby);

            numby = 2 * self.SpawnMultiplyer;
            self.steadySpawnReindeer(numby);

            amount = amount / 2;
            fairAmount = Math.Ceil(amount);

            self.gradualSpawnCheck(fairAmount);
            return;
        }
        if (self._currentWave == 15)
        {
            numby = 6 * self.SpawnMultiplyer;
            self.steadySpawnPunk(numby);

            numby = 4 * self.SpawnMultiplyer;
            self.steadySpawnReindeer(numby);

            amount = amount / 2;
            fairAmount = Math.Ceil(amount);

            self.gradualSpawnCheck(fairAmount);
            return;
        }
        if (self._currentWave == 20)
        {
            numby = 6 * self.SpawnMultiplyer;
            self.steadySpawnPunk(numby);

            numby = 4 * self.SpawnMultiplyer;
            self.steadySpawnReindeer(numby);

            numby = 1 * self.SpawnMultiplyer;
            self.steadySpawnSanta(numby);

            amount = amount / 2;
            fairAmount = Math.Ceil(amount);

            self.gradualSpawnCheck(fairAmount);
            return;
        }
        if (self.BonusRound && self._currentWave == 21)
        {
            Game.PrintAll("BONUS WAVE!");

            numby = 8 * self.SpawnMultiplyer;
            self.steadySpawnPunk(numby);

            numby = 6 * self.SpawnMultiplyer;
            self.steadySpawnReindeer(numby);

            numby = 2 * self.SpawnMultiplyer;
            self.steadySpawnSanta(numby);

            amount = amount / 2;
            fairAmount = Math.Ceil(amount);

            self.gradualSpawnCheck(fairAmount);
            return;
        }
        self.gradualSpawnCheck(amount);
    }

    function gradualSpawnCheck(amount)
    {
        if (self.GradualSpawn)
        {
            Game.SpawnTitansAsync("Default", amount);
        }
        else
        {
            Game.SpawnTitans("Default", amount);
        }
    }

    function OnNetworkMessage(sender, message)
    {
        data = Json.LoadFromString(message);
        viewID = data.Get("viewID");
        event = data.Get("event");

        titans = Game.AITitans;
        targetMatch = null;
        for (titan in titans)
        {
            if (titan.ViewID == viewID)
            {
                targetMatch = titan;
            }
        }

        if (targetMatch != null)
        {
            if (event == "CandyCanePunk")
            {
                self.ConfigureCandyCanePunk(targetMatch);
            }
            elif(event == "ReindeerCrawler")
            {
                self.ConfigureCrawlerReindeer(targetMatch);
            }
            elif(event == "Santa")
            {
                self.ConfigureSanta(targetMatch);
            }
        }
        else
        {
            # Defer handling to OnCharacterSpawn event.
            self._netbuffer.Set(viewID, event);
        }
    }

    /*
    @param titan Titan
    */
    function TryResolveBuffer(titan)
    {
        # Called on titan spawn, check if the titan has a costume to apply.
        if (!self._netbuffer.Contains(titan.ViewID))
        {
            return;
        }

        event = self._netbuffer.Get(titan.ViewID);
        if (titan != null)
        {
            if (event == "CandyCanePunk")
            {
                self.ConfigureCandyCanePunk(titan);
            }
            elif(event == "ReindeerCrawler")
            {
                self.ConfigureCrawlerReindeer(titan);
            }
            elif(event == "Santa")
            {
                self.ConfigureSanta(titan);
            }
        }

        self._netbuffer.Remove(titan.ViewID);
    }

    /*
    @param titan Titan
    */
    function TryCleanBuffer(titan)
    {
        # Called on titan death, remove the titan from the buffer.
        if (self._netbuffer.Contains(titan.ViewID))
        {
            self._netbuffer.Remove(titan.ViewID);
        }
    }

component TheChristmasTree
{
    TargetIDs = "0";
    ReplacementIDs = "0";
    TitanTargetIDs = "0";

    # Modifiers
    Health = 2500;
    HasDamageCap = false;
    DamageCap = 100;
    RecoverAfterTime = false;
    RecoverTime = 10.0;
    HasDestroyEffect = false;
    DestroyEffect = "Boom2";
    DestroyEffectScale = 5.0;
    VictimName = "Christmas Tree";
    AttacksThatCanDestroy = "";
    TeamsThatCanDestroy = "";
    Weapon = "";
    _targets = List();
    _replacements = List();
    _titanTargets = List();
    _attacksThatCanDestroy = List();
    _teamsThatCanDestroy = List();

    _isDestroyed = false;
    _isHostDestroyed = false;
    _recoverTimeLeft = 0.0;
    _health = 100;
    hasTrig = false;

    function Init()
    {
        self._targets = self.PopulateMapObjectList(self.TargetIDs);
        self._replacements = self.PopulateMapObjectList(self.ReplacementIDs);
        self._titanTargets = self.PopulateMapObjectList(self.TitanTargetIDs);

        if (self.MapObject.Active)
        {
            self.SetState(self._targets, true);
            self.SetState(self._replacements, false);
        }

        if (self.AttacksThatCanDestroy != "")
        {
            self._attacksThatCanDestroy = String.Split(self.AttacksThatCanDestroy, "/");
        }
        if (self.TeamsThatCanDestroy != "")
        {
            self._teamsThatCanDestroy = String.Split(self.TeamsThatCanDestroy, "/");
        }
        self._health = self.Health;
    }

    function GetHealth()
    {
        return self._health;
    }

    function GetMaxHealth()
    {
        return self.Health;
    }

    function PopulateMapObjectList(strList)
    {
        results = List();
        target = null;
        ssl = String.Split(strList, "/");
        for (value in ssl)
        {
            target = Map.FindMapObjectByID(Convert.ToInt(value));

            if (target != null)
            {
                results.Add(target);
            }
        }
        return results;
    }

    function SetState(objectList, state)
    {
        for (obj in objectList)
        {
            obj.Active = state;
        }
    }

    function NetworkSync(target)
    {
        message = Dict();
        message.Set("event", "sync");
        message.Set("isDestroyed", self._isHostDestroyed);
        message.Set("recoverTimeLeft", self._recoverTimeLeft);
        message.Set("health", self._health);
        if (target == null)
        {
            self.NetworkView.SendMessageAll(Json.SaveToString(message));
        }
        else
        {
            self.NetworkView.SendMessage(target, Json.SaveToString(message));
        }
    }

    function OnNetworkMessage(sender, message)
    {
        if (Network.IsMasterClient)
        {
            data = Json.LoadFromString(message);
            if (data.Get("event") == "hit")
            {
                self.OnGetHitHandler(data.Get("team"), data.Get("name"), data.Get("damage"), data.Get("type"));
                return;
            }
        }

        if (sender != Network.MasterClient)
        {
            return;
        }

        if (message == "RestoreTree")
        {
            self._health = self.Health;
            return;
        }

        data = Json.LoadFromString(message);
        hostDestroyed = data.Get("isDestroyed");
        self._recoverTimeLeft = data.Get("recoverTimeLeft");
        self._health = data.Get("health");

        if (self._isDestroyed == false && hostDestroyed == true)
        {
            self.DestroyObject();
        }
        elif(self._isDestroyed == true && hostDestroyed == false)
        {
            self.RecoverObject();
        }
    }

    function OnPlayerJoin(player)
    {
        if (Network.IsMasterClient == false)
        {
            return;
        }
        self.NetworkSync(player);
    }

    function DestroyObject()
    {
        if (self.HasDestroyEffect)
        {
            for (object in self._targets)
            {
                Game.SpawnEffect(self.DestroyEffect, object.Position, object.Rotation, self.DestroyEffectScale);
            }
        }

        self.MapObject.Active = false;
        self.SetState(self._targets, false);
        self.SetState(self._replacements, true);
        for (t in self._titanTargets)
        {
            c = t.GetComponent("TitanTarget");
            if (c != null)
            {
                c.SetEnabled(false);
            }
        }
        self._isDestroyed = true;
    }

    function RecoverObject()
    {
        self._health = self.Health;
        self.MapObject.Active = true;
        self.SetState(self._targets, true);
        self.SetState(self._replacements, false);
        for (t in self._titanTargets)
        {
            c = t.GetComponent("TitanTarget");
            if (c != null)
            {
                c.SetEnabled(true);
            }
        }
        self._isDestroyed = false;
    }

    function SignalDestroy()
    {
        self._health = 0;
        self._isHostDestroyed = true;
        self.NetworkSync(null);

        UI.SetLabelAll("MiddleCenter", "Humanity has failed to defend the tree!");
        Game.End(10.0);
    }

    function SignalRecover()
    {
        self._health = self.Health;
        self._isHostDestroyed = false;
        self.NetworkSync(null);
    }

    coroutine WaitTilRecovery()
    {
        wait self.RecoverTime;
        self.SignalRecover();
    }

    function OnGetHitHandler(team, name, damage, type)
    {
        if (Network.IsMasterClient == false)
        {
            return;
        }
        if (self._teamsThatCanDestroy.Count > 0 && self._teamsThatCanDestroy.Contains(team) == false)
        {
            return;
        }
        if (self._attacksThatCanDestroy.Count > 0 && self._attacksThatCanDestroy.Contains(type) == false)
        {
            return;
        }

        if (self.HasDamageCap)
        {
            damage = Math.Clamp(damage, 0, self.DamageCap);
        }
        self._health = self._health - damage;

        if (self._health <= 0)
        {
            self.SignalDestroy();
            if (self.RecoverAfterTime == true)
            {
                self.WaitTilRecovery();
            }
        }
    }

    function OnGetHit(character, name, damage, type)
    {
        mehName = name;
        if (!self.hasTrig && character.Type == "Titan")
        {
            self.hasTrig = true;
            Game.ShowKillScore(damage);
            Game.ShowKillFeedAll(mehName, self.VictimName, damage, self.Weapon);
        }
        self.delay();

        message = Dict();
        message.Set("event", "hit");
        message.Set("team", character.Team);
        message.Set("name", name);
        message.Set("damage", damage);
        message.Set("type", type);
        self.NetworkView.SendMessage(Network.MasterClient, Json.SaveToString(message));
    }


    coroutine delay()
    {
        wait 0.16;
        self.hasTrig = false;
    }

}
