component Paper
{
_Found = false;
function OnCollisionEnter(other)
{
if (other.Type == "Human" && other.IsMainCharacter && !self._Found)
{
self._Found = true;
other.PlaySound("Checkpoint");
Network.SendMessage(Network.MasterClient, "Found" + Convert.ToString(self.MapObject.ID));
}
}
}

class Main
{
Papers = 8;
_maxTitans = 20;
_spawnTitans = 2;
_spawnTitansEvery = 60;
_spawnTimeLeft = 60;
_allPapers = List();
_activePapers = List();
_syncTimeLeft = 0.0;
_hasSpawned = false;

function OnGameStart()
{
for (mapObject in Map.FindAllMapObjects())
{
if (String.StartsWith(mapObject.Name, "Paper"))
{
self._allPapers.Add(mapObject.ID);
mapObject.AddComponent("Paper");
if (mapObject.Static)
{
Game.Print("Warning: Paper is marked static.");
}
mapObject.Active = false;
}
}
if (Network.IsMasterClient)
{
self._allPapers.Randomize();
for (i in Range(0, self.Papers, 1))
{
if (self._allPapers.Count > i)
{
self._activePapers.Add(self._allPapers.Get(i));
}
}
self.Sync();
}
}

function OnCharacterSpawn(character)
{
if (character.IsMine)
{
if (character.Type == "Titan")
{
character.RunSpeedBase = 50;
character.RotateSpeed = 5.0;
character.AttackWait = 0;
character.AttackPause = 0.1;
character.ActionPause = 0.1;
character.AttackSpeedMultiplier = 1.5;
character.Size = 1.5;
character.CanRun = true;
character.FocusRange = 500;
}
if (character.Type == "Human")
{
character.SetSpecial("Potato");
character.MaxGas = 60;
character.CurrentGas = 60;
character.CustomDamageEnabled = true;
character.CustomDamage = 0;
}
}
if (character.Type == "Titan")
{
character.Name = "SlenderTitan";
}
}

function OnNetworkMessage(sender, message)
{
if (String.StartsWith(message, "Found") && Network.IsMasterClient)
{
id = Convert.ToInt(String.Substring(message, 5));
if (self._activePapers.Contains(id))
{
self._activePapers.Remove(id);
self.SpawnTitans();
}
self.Sync();
}
elif (String.StartsWith(message, "Sync"))
{
message = String.Substring(message, 4);
activePapers = List();
if (message != "")
{
for (idStr in String.Split(message, ","))
{
id = Convert.ToInt(idStr);
activePapers.Add(id);
}
}
for (id in self._allPapers)
{
mapObject = Map.FindMapObjectByID(id);
mapObject.Active = activePapers.Contains(id);
}
}
}

function OnSecond()
{
if (Network.IsMasterClient)
{
self._syncTimeLeft = self._syncTimeLeft - 1.0;
self._spawnTimeLeft = self._spawnTimeLeft - 1.0;
if (self._syncTimeLeft <= 0.0)
{
self.Sync();
self._syncTimeLeft = 10.0;
}
if (self._spawnTimeLeft <= 0.0)
{
self._spawnTimeLeft = self._spawnTitansEvery;
self.SpawnTitans();
}
humans = Game.Humans.Count;
playerShifters = Game.PlayerShifters.Count;
if (humans > 0 || playerShifters > 0)
{
self._hasSpawned = true;
}
if (humans == 0 && playerShifters == 0 && self._hasSpawned)
{
UI.SetLabelAll("MiddleCenter", "Humanity failed!");
Game.End(10.0);
return;
}
}
character = Network.MyPlayer.Character;
if (character != null && character.Type == "Human")
{
Camera.SetCameraMode("FPS");
}
elif (character == null)
{
Camera.ResetCameraMode();
}
}

function SpawnTitans()
{
titans = Game.Titans.Count;
if (titans < self._maxTitans)
{
Game.SpawnTitans("Normal", self._spawnTitans);
}
}

function Sync()
{
foundPapers = self.Papers - self._activePapers.Count;
if (self._activePapers.Count == 0 && !Game.IsEnding)
{
UI.SetLabelAll("MiddleCenter", "<color=#00FF00><size=30>All the notes have been found!</size></color>");
Game.End(10);
}
UI.SetLabelAll("TopCenter", "<color=red><size=30>Find all the notes</size></color>" + String.Newline + "Notes Found: " + foundPapers + " / " + self.Papers);
message = "Sync";
idStrs = List();
for (id in self._activePapers)
{
idStrs.Add(Convert.ToString(id));
}
message = message + String.Join(idStrs, ",");
Network.SendMessageAll(message);
}
}
