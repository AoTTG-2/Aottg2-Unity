class Main
{
	TitansCount = 20;
	TitansCountTooltip = "Number of titans that is spawned after the female titan died. Can be 0.";

	_hasSpawned = false;
	_hasSpawnedEnemy = false;
	_hasSpawnedTitans = false;
	_enemyShifterPosition = Vector3(0, 0, 0);

	function OnGameStart()
	{
		Game.SetPlaylist("Battle");
		Cutscene.Start("MainCutscene", true);
	}

	function OnTick()
	{
		if (Network.IsMasterClient && !Game.IsEnding)
		{
			titans = Game.Titans.Count;
			humans = Game.Humans.Count;
			aiShifters = Game.AIShifters.Count;
			playerShifters = Game.PlayerShifters.Count;
			if (humans > 0 || playerShifters > 0)
			{
				self._hasSpawned = true;
			}
			if (aiShifters > 0)
			{
				self._hasSpawnedEnemy = true;
				self._enemyShifterPosition = Game.AIShifters.Get(0).Position;
			}
			if (aiShifters == 0 && titans == 0 && self._hasSpawnedEnemy)
			{
				if (self.TitansCount > 0 && !self._hasSpawnedTitans)
				{
					self.SpawnTitans(self.TitansCount);
					self._hasSpawnedTitans = true;
					return;
				}

				UI.SetLabelAll(UILabelEnum.MiddleCenter, "Humanity wins!");
				Game.End(10.0);
				return;
			}
			if (humans == 0 && playerShifters == 0 && self._hasSpawned)
			{
				UI.SetLabelAll(UILabelEnum.MiddleCenter, "Humanity failed!");
				Game.End(10.0);
				return;
			}
			UI.SetLabelAll(UILabelEnum.TopCenter, "Titans Left: " + Convert.ToString(titans + aiShifters));
		}
	}

	coroutine SpawnTitans(count)
	{
		i = 0;
		titans = List();
		while (i < count)
		{
			titan = Game.SpawnTitan(TitanTypeEnum.Default);
			titan.DetectRange = 0;
			titan.FocusRange = 0;
			titans.Add(titan);

			i += 1;
			wait 0.01;
		}

		for(t in titans)
		{
			t.MoveTo(self._enemyShifterPosition, 1, true);
		}

		wait 13.0;

		for(t in titans)
		{
			if (t != null)
			{
				t.Wander();
				t.DetectRange = 2000;
				t.FocusRange = 2000;
			}
		}
	}
}
