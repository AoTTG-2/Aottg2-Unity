class Main
{
    DebugMode = false;
    RespawnMode = "OnCapture";
    ReinforcementCallsEnabled = true;
    MaxReinforcementCalls = 4;

    RespawnModeDropbox = "OnCapture, None";

    _debug = null;
    _eventController = null;
    _enabled = false;

    _reinformentCalls = 0;

    function OnGameStart()
    {
        ReinforcementZones.Setup(self);
        
        if (!self.ReinforcementCallsEnabled)
        {
            ReinforcementZones.DeactivateAll();
        }

        if (!Network.IsMasterClient)
        {
            return;
        }

        if (self.ReinforcementCallsEnabled)
        {
            InGameUI.UpdateReinforcementCalls(0, self.MaxReinforcementCalls);
        }

        self._debug = Debug(self.DebugMode);
        self._eventController = EventController(self);
        self._enabled = true;
    }

    function OnCharacterSpawn(character)
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._eventController.OnCharacterSpawn(character);
        }
    }

    function OnCharacterDie(victim, killer, killerName)
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._eventController.OnCharacterDie(victim, killer, killerName);
        }
    }

    function OnCharacterDamaged(victim, killer, killerName, damage)
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._eventController.OnCharacterDamaged(victim, killer, killerName, damage);
        }
    }

    function OnPlayerLeave(player)
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self.LoseIfPlayersAreDead();
        }
    }

    coroutine OnPlayerSpawn(player, character)
    {
        if (self._enabled && Network.IsMasterClient && character.Type == "Titan")
        {
            wait 1.0;
            character.GetKilled("Server");
            Network.SendMessage(player, "pt");
        }
    }

    function OnNetworkMessage(sender, message)
    {
        if (message == "pt")
        {
            Game.Print("<size=18>Player titans are not supported</size>");
        }
        else
        {
            data = Json.LoadFromString(message);
            msgType = data.Get("type");
            if (msgType == "Dialogue")
            {
                key = data.Get("key");
                profileIndex = data.Get("profileIndex");
                DialogueUtils.ShowDialogue(key, profileIndex);
            }
            elif (msgType == "RevealTitans")
            {
                duration = data.Get("duration");
                TitanUtils.RevealTitans(duration);
            }
            elif (msgType == "Gate")
            {
                state = data.Get("state");
                interactable = data.Get("interactable");

                wallGateObj = Map.FindMapObjectByTag("StartWallGate");
                wallGate = wallGateObj.GetComponent("WallGateA1");
                wallGate.Interactable = interactable;
                wallGate.SetState(state);
            }
            elif (msgType == "Reinforcement")
            {
                pos = data.Get("pos");
                zone = CaptureZones.GetNearestCaptureZone(pos);
                self.Reinforcement(zone);
            }
            elif (msgType == "UI")
            {
                msg = data.Get("msg");
                if (msg == "lives")
                {
                    current = data.Get("current");
                    max = data.Get("max");
                    InGameUI.UpdateLives(current, max);
                }
                else
                {
                    current = data.Get("current");
                    InGameUI.UpdateRespawnTime(current);
                }
            }
            else
            {
                Game.Print(data);
            }
        }
    }

    function OnFrame()
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._debug.OnFrame();
            self._eventController.OnFrame();
        }
    }

    function OnSecond()
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._debug.OnSecond();
            self._eventController.OnSecond();
        }
    }

    function OnTick()
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._eventController.OnTick();
        }
    }

    function OnChatInput(message)
    {
        if (self._enabled && Network.IsMasterClient)
        {
            self._debug.OnChatInput(message);
        }
    }

    function HandleInteractAction(trigger)
    {
        position = trigger.MapObject.Position;
        NetworkMessage().Set("type", "Reinforcement").Set("pos", position).Send(Network.MasterClient);
    }

    function Reinforcement(captureZone)
    {
        if (self._reinformentCalls < self.MaxReinforcementCalls && PlayerUtils.AnyPlayerDead())
        {
            PlayerUtils.RevivePlayers(captureZone);
            InGameUI.UpdateReinforcementCalls(self._reinformentCalls, self.MaxReinforcementCalls);
            self._reinformentCalls += 1;
        }
    }

    function LoseIfPlayersAreDead()
    {
        alive = false;
        i = 0;
        while (i < Network.Players.Count)
        {
            p = Network.Players.Get(i);
            if (p.CharacterType == "Human" && p.Status == "Alive")
            {
                alive = true;
                i = Network.Players.Count;
            }
            i += 1;
        }

        if (!alive)
        {
            UI.SetLabelAll("MiddleCenter", "Humanity failed!");
			Game.End(10.0);
        }
    }
}

    function OnCaptured(captureZone, newOwner)
    {

        for(e in self._events)
        {
            e.OnCaptured(captureZone, newOwner);
        }

        if (newOwner == "Titan")
        {
            if (captureZone.Order == self._currentSpawnPointIndex)
            {
                self.MoveSpawnPointsBack();
            }

            # Titans have captured a zone previously captured by humans
            if (self._capturedZones.Contains(captureZone))
            {
                self._capturedZones.Remove(captureZone);
            }
        }
        elif (newOwner == "Human")
        {
            c = CaptureZones.Get(self._currentIndex);

            if (captureZone.Order > self._currentSpawnPointIndex)
            {
                PlayerUtils.SetSpawnPointInsideCaptureZone(c);

                if (self._main.RespawnMode == "OnCapture")
                {
                    PlayerUtils.RevivePlayers(c);
                }
                
                self._currentSpawnPointIndex = captureZone.Order;
            }

            self._capturedZones.Add(c);
            if (!self.TryStartNextEvent())
            {
                # If the zone is captured and the titans are remaining
                # show the dialogue and "reveal" the titans
                if (EventUtils.IsTown(self._currentIndex))
                {
                    DialogueUtils.SendShowDialogueRandomProfileMessage("TownCapturedTitansRemain");
                }
                elif (self._currentIndex == 3)
                {
                    DialogueUtils.SendShowDialogueRandomProfileMessage("BeforeAnnie1");
                }
                else
                {
                    DialogueUtils.SendShowDialogueRandomProfileMessage("PointCapturedTitansRemain");
                }
                EventUtils.RevealTitans(self._events.Get(self._currentIndex), null);
            }
        }
    }

    function TryStartNextEvent()
    {
        inBounds = self._currentIndex < self._events.Count;

        if (inBounds && CaptureZones.CheckIfPreviousZonesAreCapturedByHumans(self._currentIndex))
        {
            capturedByHumans = CaptureZones.IsZoneCapturedByHumans(self._currentIndex);
            titansAreDead = Game.Titans.Count == 0;
            if (capturedByHumans && titansAreDead && self._events.Get(self._currentIndex).CanActivateNextCaptureZone())
            {
                prevIndex = self._currentIndex;
                self._events.Get(self._currentIndex).OnEventEnd();
                CaptureZones.DisableCaptureZone(self._currentIndex);
                self._currentIndex = self._currentIndex + 1;

                if (self._currentIndex < CaptureZones.Count())
                {
                    self._events.Get(self._currentIndex).OnEventStart();
                    CaptureZones.ActivateCaptureZone(self._currentIndex);

                    if (EventUtils.IsTown(prevIndex))
                    {
                        DialogueUtils.SendShowDialogueRandomProfileMessage("TownCaptured");
                    }
                    else
                    {
                        DialogueUtils.SendShowDialogueRandomProfileMessage("PointCaptured");
                    }
                }
                else
                {
                    # Captured all zones
                    CaptureZones.DisableAllCaptureZones();
                }
                return true;
            }
        }

        return false;
    }

    function OnCapturing(captureZone, capturer)
    {
        for(e in self._events)
        {
            e.OnCapturing(captureZone, capturer);
        }
    }

    function OnContested(captureZone)
    {
        if (captureZone.Order == self._currentSpawnPointIndex)
        {
            self.MoveSpawnPointsBack();
        }
    }

    function OnCharacterDie(victim, killer, killerName)
    {
        for(e in self._events)
        {
            e.OnCharacterDie(victim, killer, killerName);
        }

        if (victim.Type == "Human")
        {
            self._main.LoseIfPlayersAreDead();
        }
        elif (victim.Type == "Titan")
        {
            Utils.TryRemoveListElement(self._titans, victim);
            self.TryStartNextEvent();
        }
    }

    function OnCharacterSpawn(character)
    {
        if (character != null && character.Type == "Titan")
        {
            self._titans.Add(character);
        }
    }

    function OnCharacterDamaged(victim, killer, killerName, damage)
    {
        for(e in self._events)
        {
            e.OnCharacterDamaged(victim, killer, killerName, damage);
        }
    }

    function MoveSpawnPointsBack()
    {
        foundSpawnPoint = false;
        i = self._currentSpawnPointIndex;
        while (i > 0)
        {
            prev = CaptureZones.Get(i - 1);
            if (prev.GetState() == "Captured" && prev.GetCurrentOwner() == "Human")
            {
                self._currentSpawnPointIndex = i - 1;
                PlayerUtils.SetSpawnPointInsideCaptureZone(prev);
                foundSpawnPoint = true;
                i = -1;
            }
            i -= 1;
        }

        if (!foundSpawnPoint)
        {
            PlayerUtils.SetSpawnPointInsideCaptureZone(null);
            self._currentSpawnPointIndex = -1;
        }
    }

    function OnFrame()
    {
        for(e in self._events)
        {
            e.OnFrame();
        }

        if (self._currentIndex < CaptureZones.Count())
        {
            self._events.Get(self._currentIndex).OnEventUpdate();
        }
    }

    function OnSecond()
    {
        if (self._currentIndex >= CaptureZones.Count())
        {
            return;
        }

        captureZone = CaptureZones.Get(self._currentIndex);
        self._humansInCaptureZone = captureZone.GetPlayersOfTeamInZone("Human");
        for(titan in self._titans)
        {
            /* 
             * If the titan doesn't have a target
             * and if there are human players in the zone
             * select one randomly and attack him
             * but if there is no one in the zone, move towards the zone
             */

            if (self.HasTarget(titan) == false)
            {
                if (self._humansInCaptureZone != null)
                {
                    target = Utils.GetRandomListElement(self._humansInCaptureZone);
                    self._titansTargetMap.Set(titan, target);
                    titan.Target(target, 9999);
                }
                else
                {
                    pos = captureZone.MapObject.Position;
                    dist = Vector3.Distance(titan.Position, pos);
                    if (dist > 150) # allow titan to move a little outside the zone
                    {
                        titan.MoveTo(pos, 100, false);
                    }
                }
            }
            else
            {
                /* 
                 * If the titan has a target
                 * and if the target is not alive anymore
                 * or the target is outside the capture zone, remove the target
                 * so the titan can search for a new one
                 */

                target = self._titansTargetMap.Get(titan);
                
                if (target == null || self._humansInCaptureZone == null || !self._humansInCaptureZone.Contains(target))
                {
                    self._titansTargetMap.Remove(titan);
                }
            }
        }

        for(e in self._events)
        {
            e.OnSecond();
        }
    }

    function HasTarget(titan)
    {
        if (self._titansTargetMap.Contains(titan))
        {
            target = self._titansTargetMap.Get(titan);
            return target != null;
        }

        return false;
    }

    function OnTick()
    {
        for(e in self._events)
        {
            e.OnTick();
        }
    }
}

        if (self._isCapturedByHumans)
        {
            return;
        }

        if (self._isHumanCapturing)
        {
            # If humans made some progress, start spawning titans continuously.
            # should probably check for how long the humans have been in the zone
            # instead of checking if they made atleast 10% progress
            self._humanCaptureProgress = self.CaptureZone.GetCaptureProgress();

    {

        {
            if (self._fastForwarding)
            {
                Time.TimeScale = 1.0;

            }
            else
            {
                Time.TimeScale = self._timeScale;
                self._fastForwarding = true;
            }
        }
        elif (Input.GetKeyDown("Interaction/Function3"))
        {
            if (Network.MyPlayer.Status != "Alive")
            {
                return;
            }

            position = Network.MyPlayer.Character.Position;
            zone = self.GetNearestCaptureZone(position);
            sp = TitanSpawnPoints.GetSpawnPoint(zone);
            titan = Game.SpawnTitanAt("Abnormal", sp);
            titan.FocusRange = 0;
            titan.DetectRange = 0;
            titan.MoveTo(zone.MapObject.Position, 10, false);
        }
    }

    function GetNearestCaptureZone(position)
    {
        nearest = null;
        nearestDist = 99999.0;
        for(zone in CaptureZones.GetAll())
        {
            dist = Vector3.Distance(position, zone.MapObject.Position);
            if (dist < nearestDist)
            {
                nearest = zone;
                nearestDist = dist;
            }
        }

        return nearest;
    }

    function OnSecond()
    {
        if (!self.Enabled)
        {
            return;
        }

        player = Network.MyPlayer;
        if (player.Status == "Alive")
        {
            character = player.Character;
            if (character != null && character.Type == "Human")
            {
                character.MaxHealth = 1000;
                character.Health = 1000;
                character.RefillImmediate();
            }
        }
    }

    function OnChatInput(message)
    {
        if (!self.Enabled)
        {
            return;
        }

        if (String.StartsWith(message, "--xyz"))
        {
            pos = Network.MyPlayer.Character.Position;
            Game.Print(pos);
        }

        if (String.StartsWith(message, "--kill"))
        {
            for(p in Network.Players)
            {
                if (p.Status == "Alive")
                {
                    p.Character.GetKilled("Server");
                }
            }
        }

        if (String.StartsWith(message, "--tscale"))
        {
            str = String.Split(message, " ");
            self._timeScale = Convert.ToFloat(str.Get(1));
        }
    }

extension CaptureZones
{
    _captureZones = List();

    function Setup(callbackReceiver)
    {
        captureZones = Map.FindMapObjectsByName("CaptureZone1");
        captureZoneOrderMap = Dict();
        for(captureZone in captureZones)
        {
            captureZoneComponent = captureZone.GetComponent("CaptureZone");
            captureZoneComponent.SetupCallbackReceiver(callbackReceiver);
            if (captureZoneOrderMap.Contains(captureZoneComponent.Order) == true)
            {
                Game.Print("Capture zone order " + captureZoneComponent.Order + " is already added, please remove duplicate zone!");
                Game.End(10);
                return;
            }
            captureZoneOrderMap.Set(captureZoneComponent.Order, captureZoneComponent);
        }

        keys = captureZoneOrderMap.Keys;
        keys.Sort();
        for(key in keys)
        {
            self._captureZones.Add(captureZoneOrderMap.Get(key));
        }
    }

    function Count()
    {
        return self._captureZones.Count;
    }

    function GetAll()
    {
        return self._captureZones;
    }

    function Get(index)
    {
        return self._captureZones.Get(index);
    }

    function IsZoneCapturedByHumans(captureZone)
    {
        if (Convert.IsInt(captureZone))
        {
            return self.Get(captureZone).GetCurrentOwner() == "Human";
        }

        return captureZone.GetCurrentOwner() == "Human";
    }

    function CheckIfPreviousZonesAreCapturedByHumans(currentIndex)
    {
        previousZonesAreCaptured = true;
        if (currentIndex > 0)
        {
            i = currentIndex;
            while (i >= 0)
            {
                if (!self.IsZoneCapturedByHumans(i))
                {
                    previousZonesAreCaptured = false;
                    i = -1; # break
                }
                i -= 1;
            }
        }
        return previousZonesAreCaptured;
    }

    function GetNearestCaptureZone(position)
    {
        nearest = null;
        nearestDist = 99999.0;
        for(zone in self._captureZones)
        {
            dist = Vector3.Distance(position, zone.MapObject.Position);
            if (dist < nearestDist)
            {
                nearest = zone;
                nearestDist = dist;
            }
        }

        return nearest;
    }

    function DeactivateAllCaptureZones()
    {
        for(cz in self._captureZones)
        {
            cz.MapObject.Active = false;
        }
    }

    function ActivateCaptureZone(index)
    {
        self.Get(index).MapObject.Active = true;
    }

    function DisableAllCaptureZones()
    {
        for(i in Range(0, self.Count(), 1))
        {
            self.DisableCaptureZone(i);
        }
    }

extension TitanSpawnPoints
{
    _spawnPointsMap = Dict();
    _annieSpawnPoint4 = null;
    _annieSpawnPoint8 = null;

    function Setup(captureZones)
    {
        self._spawnPointsMap.Set(captureZones.Get(0), Map.FindMapObjectsByTag("TitanSpawnPoint1"));
        self._spawnPointsMap.Set(captureZones.Get(1), Map.FindMapObjectsByTag("TitanSpawnPoint2"));
        self._spawnPointsMap.Set(captureZones.Get(2), Map.FindMapObjectsByTag("TitanSpawnPoint3"));
        self._spawnPointsMap.Set(captureZones.Get(3), Map.FindMapObjectsByTag("TitanSpawnPoint4"));
        self._spawnPointsMap.Set(captureZones.Get(4), Map.FindMapObjectsByTag("TitanSpawnPoint5"));
        self._spawnPointsMap.Set(captureZones.Get(5), Map.FindMapObjectsByTag("TitanSpawnPoint6"));
        self._spawnPointsMap.Set(captureZones.Get(6), Map.FindMapObjectsByTag("TitanSpawnPoint7"));
        self._spawnPointsMap.Set(captureZones.Get(7), Map.FindMapObjectsByTag("TitanSpawnPoint8"));

        self._annieSpawnPoint4 = Map.FindMapObjectsByTag("AnnieSpawn1").Get(0);
        self._annieSpawnPoint8 = Map.FindMapObjectsByTag("AnnieSpawn2").Get(0);
    }

    function GetSpawnPoint(captureZone)
    {
        if (self._spawnPointsMap.Contains(captureZone))
        {
            sp = Utils.GetRandomListElement(self._spawnPointsMap.Get(captureZone));
            return sp.Position;
        }

        if (captureZone.Order == 0)
        {
            pos = self.CaptureZone.MapObject.Position;
            angle = 90 - Math.Rad2Deg(Random.RandomFloat(0, Math.PI));
            position = Vector3(Math.Sin(angle), 0, Math.Cos(angle)) * Random.RandomFloat(minRadius, maxRadius);
            position += pos;
            return position;
        }

        return captureZone.MapObject.Position;
    }

    function GetAnnieSpawnPoint(captureZone)
    {
        if (captureZone.Order == 3 && self._annieSpawnPoint4 != null)
        {
            return self._annieSpawnPoint4.Position;
        }

        if (captureZone.Order == 7 && self._annieSpawnPoint8 != null)
        {
            return self._annieSpawnPoint8.Position;
        }

        return captureZone.MapObject.Position;
    }
}

extension PlayerUtils
{
    function AnyPlayerDead()
    {
        dead = false;
        for(p in Network.Players)
        {
            if (p.Status != "Alive")
            {
                dead = true;
            }
        }

        return dead;
    }

    coroutine RevivePlayers(captureZone)
    {
        wait 1.0;
        self.SetSpawnPointInsideCaptureZone(captureZone);
        Game.SpawnPlayerAll(false);
    }

    function SetSpawnPointInsideCaptureZone(captureZone)
    {
        if (captureZone == null)
        {
            for(p in Network.Players)
            {
                p.SpawnPoint = null;
            }
            return;
        }

        yPos = 10.0;
        zoneRadius = 10.0;

        for(p in Network.Players)
        {
            position = Utils.RandomPointInsideUnitCircleXZ();
            position *= zoneRadius;
            position.Y = yPos;
            position += captureZone.MapObject.Position;
            p.SpawnPoint = position;
        }
    }

    function GetPlayerById(id)
    {
        for(p in Network.Players)
        {
            if (p.ID == id)
            {
                return p;
            }
        }

        return null;
    }
}
extension TitanUtils

    function RevealTitans(seconds)
    {
        for(t in Game.Titans)

            if (t != null)
            {
                if (seconds == null)

                    t.AddOutline();

                }
            }
        }
    }

    function SpawnAnnie(health, captureZone)
    {
        position = TitanSpawnPoints.GetAnnieSpawnPoint(captureZone);

        shifter = Game.SpawnShifterAt("Annie", position);
        shifter.MaxHealth = health;
        shifter.Health = health;

    {
        i = 0;

        while (i < count)
        {
            position = TitanSpawnPoints.GetSpawnPoint(captureZone);
            titan = Game.SpawnTitanAt("Default", position);

            if (Math.Mod(i, 10) == 0)
            {
                wait 0.2;
            }

            i += 1;
        }
    }
}
extension Utils
{
    function RandomPointInsideUnitCircleXZ()
    {
        angle = Math.Rad2Deg(Random.RandomFloat(0.0, 2 * Math.PI));
        position = Vector3(Math.Sin(angle), 0, Math.Cos(angle));
        return position;
    }

    function GetRandomListElement(list)
    {
        i = Random.RandomInt(0, list.Count);
        return list.Get(i);
    }

    function IndexOfListElement(list, element)
    {
        i = 0;
        while (i < list.Count)
        {
            if (list.Get(i) == element)
            {
                return i;
            }

            i += 1;
        }

        return -1;
    }

    function TryRemoveListElement(list, element)
    {
        index = self.IndexOfListElement(list, element);
        if (index != -1)
        {
            list.RemoveAt(index);
            return true;
        }

        return false;
    }
}
